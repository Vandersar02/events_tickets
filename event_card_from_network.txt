import 'package:flutter/material.dart';

class EventCard extends StatelessWidget {
  final String date;
  final String title;
  final String location;
  final String imageUrl;
  final bool isFree;
  final int attendeesCount;

  const EventCard({
    super.key,
    required this.date,
    required this.title,
    required this.location,
    required this.imageUrl,
    this.isFree = false,
    this.attendeesCount = 0,
  });

  @override
  Widget build(BuildContext context) {
    return Card(
      shape: RoundedRectangleBorder(
        borderRadius: BorderRadius.circular(16.0),
      ),
      elevation: 8.0,
      shadowColor: Colors.blue.withOpacity(0.2), // Couleur bleue
      child: Padding(
        padding: const EdgeInsets.all(8.0),
        child: Column(
          crossAxisAlignment: CrossAxisAlignment.start,
          children: [
            Stack(
              children: [
                // Image de l'événement
                ClipRRect(
                  borderRadius: BorderRadius.circular(12.0),
                  child: 
                  Image.network(
                    imageUrl,
                    height: 120.0,
                    width: double.infinity,
                    fit: BoxFit.cover,
                  ),
                ),
                // Date dans le coin supérieur gauche
                Positioned(
                  top: 10.0,
                  left: 10.0,
                  child: Container(
                    padding: const EdgeInsets.symmetric(
                        vertical: 4.0, horizontal: 8.0),
                    decoration: BoxDecoration(
                      color: Colors.white,
                      borderRadius: BorderRadius.circular(8.0),
                    ),
                    child: Text(
                      date,
                      style: const TextStyle(
                        color: Colors.redAccent,
                        fontWeight: FontWeight.bold,
                      ),
                    ),
                  ),
                ),
                // Étiquette "Free" si l'événement est gratuit
                if (isFree)
                  Positioned(
                    top: 10.0,
                    right: 10.0,
                    child: Container(
                      padding: const EdgeInsets.symmetric(
                          vertical: 4.0, horizontal: 8.0),
                      decoration: BoxDecoration(
                        color: Colors.green,
                        borderRadius: BorderRadius.circular(8.0),
                      ),
                      child: const Text(
                        'Free',
                        style: TextStyle(
                          color: Colors.white,
                          fontWeight: FontWeight.bold,
                        ),
                      ),
                    ),
                  ),
              ],
            ),
            const SizedBox(height: 10.0),
            // Titre de l'événement
            Text(
              title,
              maxLines: 2,
              overflow: TextOverflow.ellipsis,
              style: const TextStyle(
                fontSize: 16.0,
                fontWeight: FontWeight.bold,
              ),
            ),
            const SizedBox(height: 5.0),
            // Liste des avatars des participants et nombre d'autres participants
            Row(
              children: [
                // Affichage des avatars des participants
                SizedBox(
                  width: 60.0,
                  child: Stack(
                    children: List.generate(
                      3, // Afficher un maximum de 3 avatars
                      (index) => Positioned(
                        left: index * 20.0, // Décalage horizontal des avatars
                        child: const CircleAvatar(
                          radius: 12.0,
                          backgroundImage: NetworkImage(
                            'https://via.placeholder.com/150', // Placeholder avatar URL
                          ),
                        ),
                      ),
                    ),
                  ),
                ),
                const SizedBox(width: 8.0),
                // Indicateur du nombre de participants supplémentaires
                Text(
                  '+$attendeesCount Going',
                  style: const TextStyle(
                    color: Colors.blue, // Couleur bleue
                    fontWeight: FontWeight.bold,
                  ),
                ),
              ],
            ),
            const SizedBox(height: 5.0),
            // Adresse de l'événement
            Row(
              children: [
                const Icon(
                  Icons.location_on,
                  color: Colors.grey,
                  size: 16.0,
                ),
                const SizedBox(width: 4.0),
                Text(
                  location,
                  style: const TextStyle(
                    color: Colors.grey,
                    fontSize: 14.0,
                  ),
                ),
              ],
            ),
          ],
        ),
      ),
    );
  }
}
